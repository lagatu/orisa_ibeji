Overall Health Dataset (Databricks SQL)

NOTE:
Confirm the following before production deployment:
1. Actual uniqueness/business key.
2. Actual pipeline execution log table.
3. Whether Overall Health is a simple average or weighted average.

SQL:

WITH completeness AS
(
    SELECT
        ROUND(
            (
                COUNT(*) -
                SUM(CASE
                        WHEN PortfolioCode IS NULL OR TRIM(PortfolioCode) = ''
                        THEN 1 ELSE 0
                    END)
            ) * 100.0 / COUNT(*),
            2
        ) AS score
    FROM silver_dev.pmg_r_vpositionscurrent
),

timeliness AS
(
    SELECT
        CASE
            WHEN EXISTS
            (
                SELECT 1
                FROM etl.PipelineExecutionLog
                WHERE PipelineName = 'federated_acquisition_pipeline'
                  AND Status = 'Succeeded'
                  AND CAST(StartTime AS DATE) = CURRENT_DATE() - 1
            )
            THEN 100.00
            ELSE 0.00
        END AS score
),

uniqueness AS
(
    SELECT
        ROUND(
            (t.TotalRecords - f.FailedRecords) * 100.0 / t.TotalRecords,
            2
        ) AS score
    FROM
        (
            SELECT COUNT(*) AS TotalRecords
            FROM gold_dev.pmg.factpositionsdaily
        ) t
    CROSS JOIN
        (
            SELECT COALESCE(SUM(Cnt - 1), 0) AS FailedRecords
            FROM
            (
                SELECT COUNT(*) AS Cnt
                FROM gold_dev.pmg.factpositionsdaily
                GROUP BY PortfolioCode, FINAccountNumber, PositionDate
                HAVING COUNT(*) > 1
            ) d
        ) f
)

SELECT
    ROUND(AVG(score), 2) AS OverallHealth
FROM
(
    SELECT score FROM completeness
    UNION ALL
    SELECT score FROM timeliness
    UNION ALL
    SELECT score FROM uniqueness
) dq;
